    53тЖТ    53тЖТfunction formatRelativeTime(dateStr: string) {
    54тЖТ    54тЖТ  const diff = Date.now() - new Date(dateStr).getTime();
    55тЖТ    55тЖТ  const mins = Math.floor(diff / 60000);
    56тЖТ    56тЖТ  if (mins < 60) return `${mins}m ago`;
    57тЖТ    57тЖТ  const hrs = Math.floor(mins / 60);
    58тЖТ    58тЖТ  if (hrs < 24) return `${hrs}h ago`;
    59тЖТ    59тЖТ  const days = Math.floor(hrs / 24);
    60тЖТ    60тЖТ  if (days < 30) return `${days}d ago`;
    61тЖТ    61тЖТ  return new Date(dateStr).toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
    62тЖТ    62тЖТ}
